346 class Publisher : public BaseMsgbusThread {
347 private:
348  // Publisher context
349  publisher_ctx_t* m_pub_ctx;
350 
351  // Input message queue
352  MessageQueue* m_input_queue;
353 
354  // AppName variable
355  std::string m_service_name;
356 
357  // Profiling variable
358  Profiling* m_profile;
359 
360  Publisher& operator=(const Publisher& src) {
361  return *this;
362  }
363 
364 protected:
368  void run() override {
369  LOG_DEBUG_0("Publisher thread started");
370 
371  // Duration to wait in between checking if the publisher should stop
372  auto duration = std::chrono::milliseconds(250);
373  Serializable* msg = NULL;
374  msg_envelope_t* env = NULL;
375  msgbus_ret_t ret = MSG_SUCCESS;
376 
377  // Profiling related variables
378  int64_t serilization_start= 0;
379  std::string serialization_start_ts_str = m_service_name + "_serialization_entry";
380  std::string serialization_end_ts_str = m_service_name + "_serialization_exit";
381  std::string publisher_ts_str = m_service_name + "_publisher_ts";
382 
383  while(!m_stop.load()) {
384  if(m_input_queue->wait_for(duration)) {
385  // Pop the envelope off the top of the queue
386  msg = m_input_queue->pop();
387  if(msg == NULL) {
388  LOG_WARN_0("Got NULL serializable message...");
389  continue;
390  }
391 
392  try {
393  // Serialize message into a message envelope
394  if(this->m_profile->is_profiling_enabled()) {
395  // Getting the current timestamp since DO_PROFILNG can't be used
396  // with msg
397  serilization_start = this->m_profile->get_curr_time_as_int_epoch();
398  env = msg->serialize();
399  DO_PROFILING(this->m_profile, env, serialization_end_ts_str.c_str());
400  } else {
401  env = msg->serialize();
402  }
403  if(env == NULL) {
404  delete msg;
405  msg = NULL;
406  LOG_ERROR_0("Failed to serialize message to msg envelope");
407  msgbus_msg_envelope_destroy(env);
408  m_err_cv.notify_all();
409  break;
410  }
411 
412  if(this->m_profile->is_profiling_enabled()) {
413  // Adding the obtained start serialization timestamp to meta-data
414  msg_envelope_elem_body_t* serialization_time_start = msgbus_msg_envelope_new_integer(serilization_start);
415  if (serialization_time_start == NULL) {
416  LOG_ERROR_0("Failed to create profiling timestamp element");
417  msgbus_msg_envelope_destroy(env);
418  m_err_cv.notify_all();
419  break;
420  }
421  ret = msgbus_msg_envelope_put(env, serialization_start_ts_str.c_str(), serialization_time_start);
422  if(ret != MSG_SUCCESS) {
423  LOG_ERROR_0("Failed to add profiling timestamp to message envelope...");
424  msgbus_msg_envelope_destroy(env);
425  m_err_cv.notify_all();
426  break;
427  }
428  DO_PROFILING(this->m_profile, env, publisher_ts_str.c_str());
429  }
430 
431  // Publish message
432  ret = msgbus_publisher_publish(m_ctx, m_pub_ctx, env);
433  if(ret != MSG_SUCCESS) {
434  LOG_ERROR_0("Failed to publish message...");
435  msgbus_msg_envelope_destroy(env);
436  m_err_cv.notify_all();
437  break;
438  }
439 
440  LOG_DEBUG("Message published successfully");
441 
442  // Clean up after publication attempt...
443  msgbus_msg_envelope_destroy(env);
444  } catch(const std::exception& e) {
445  LOG_ERROR("Failed to serialize message: %s", e.what());
446  delete msg;
447  msg = NULL;
448  m_err_cv.notify_all();
449  break;
450  }
451  }
452  }
453 
454  LOG_DEBUG_0("Publisher thread stopped");
455  };

496 template<class T>
497 class Subscriber : public BaseMsgbusThread {
498 private:
499  // Subscriber context
500  recv_ctx_t* m_recv_ctx;
501 
502  // Output message queue
503  MessageQueue* m_output_queue;
504 
505  // Timeout for time to wait between receiving and checking if the stop
506  // signal has been sent
507  int m_timeout;
508 
509  // AppName variable
510  std::string m_service_name;
511 
512  // Profiling variable
513  Profiling* m_profile;
514 
515  Subscriber& operator=(const Subscriber& src) {
516  return *this;
517  }
518 
519 protected:
523  void run() override {
524  LOG_DEBUG_0("Subscriber thread started");
525 
526  msg_envelope_t* msg = NULL;
527  msgbus_ret_t ret = MSG_SUCCESS;
528 
529  // Profiling related variables
530  std::string subscriber_ts = m_service_name + "_subscriber_ts";
531  std::string subscriber_exit_ts = m_service_name + "_subscriber_exit_ts";
532  std::string subscriber_blocked_ts = m_service_name + "_subscriber_blocked_ts";
533 
534  while(!m_stop.load()) {
535  ret = msgbus_recv_timedwait(m_ctx, m_recv_ctx, m_timeout, &msg);
536  if(ret == MSG_SUCCESS) {
537  // Received message
538  T* received = new T(msg);
539 
540  // Add timestamp after message is received
541  // if profiling is enabled
542  if(this->m_profile->is_profiling_enabled()) {
543  DO_PROFILING(this->m_profile, msg, subscriber_ts.c_str());
544  }
545  QueueRetCode ret_queue = m_output_queue->push(received);
546  if(ret_queue == QueueRetCode::QUEUE_FULL) {
547  // Add timestamp which acts as a marker if queue if blocked
548  if(this->m_profile->is_profiling_enabled()) {
549  DO_PROFILING(this->m_profile, msg, subscriber_blocked_ts.c_str());
550  }
551  if(m_output_queue->push_wait(received) != QueueRetCode::SUCCESS) {
552  LOG_ERROR_0("Failed to enqueue received message, "
553  "message dropped");
554  delete received;
555  msgbus_msg_envelope_destroy(msg);
556  m_err_cv.notify_all();
557  break;
558  } else {
559  // Dropping pointer to message here because the memory for
560  // the envelope is not owned by the received variable
561  msg = NULL;
562  }
563  }
564  // Add timestamp for subscriber exit
565  if(this->m_profile->is_profiling_enabled()) {
566  DO_PROFILING(this->m_profile, msg, subscriber_exit_ts.c_str());
567  }
568  } else if(ret != MSG_RECV_NO_MESSAGE) {
569  LOG_ERROR("Error receiving message: %d", ret);
570  m_err_cv.notify_all();
571  break;
572  }
573  }
574 
575  LOG_DEBUG_0("Subscriber thread stopped");
576  };
